pipeline {
    agent any

    options {
        timestamps()
    }

    stages {

        stage('Checkout Application Code') {
            steps {
                git branch: 'main',
                    url: 'https://github.com/Daniyal1Hazari/my-repository.git'
            }
        }

        stage('Build Application') {
            steps {
                echo "Starting Build Process..."

                // Replace with actual build command (maven/npm/gradle etc.)
                sh 'echo "Building application..."'
                sh 'ls -la'
            }
        }
    }

    post {
        success {
            echo "job1 completed successfully. Triggering job2..."

            // This triggers job2 and waits for it to complete
            build job: 'job2', wait: true
        }

        failure {
            echo "job1 failed. job2 will NOT be triggered."
        }
    }
}
