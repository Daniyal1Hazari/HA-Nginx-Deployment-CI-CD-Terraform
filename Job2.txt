pipeline {
    agent any

    options {
        timestamps()
    }

    stages {

        stage('Checkout Code') {
            steps {
                git branch: 'main',
                    url: 'https://github.com/Daniyal1Hazari/my-repository.git'
            }
        }

        stage('Terraform Init') {
            steps {
                dir('terraform') {
                    sh 'terraform init -input=false'
                }
            }
        }
    }

    post {
        success {
            echo "job2 successful. Triggering job3..."

            // Trigger next job only if this job succeeds
            build job: 'job3', wait: true
        }

        failure {
            echo "Terraform Init failed. job3 will NOT run."
        }
    }
}
